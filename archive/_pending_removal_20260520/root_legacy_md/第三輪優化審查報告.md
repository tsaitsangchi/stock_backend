```python
PID_DIR = Path("scripts/outputs/pids")
PID_DIR.mkdir(exist_ok=True)

def get_running_trains() -> list[str]:
    """讀取 PID 檔案，確認進程仍在運行"""
    running = []
    for pid_file in PID_DIR.glob("*.pid"):
        sid = pid_file.stem
        pid = int(pid_file.read_text().strip())
        try:
            os.kill(pid, 0)  # 發送 signal 0，只檢查進程是否存在
            running.append(sid)
        except ProcessLookupError:
            pid_file.unlink()  # 清理殭屍 PID 檔案
    return running

def launch_training(sid: str, cmd: list):

    (PID_DIR / f"{sid}.pid").write_text(str(proc.pid))
```

### 3.3 kwave_score 特徵的全系統不一致性（三輪均存在）

以下位置都使用了 `kwave_score`，但沒有任何一個地方確認了它的計算來源：

| 使用位置 | 用途 |
|---|---|
| `signal_filter.py` | 動態調整機率門檻 |
| `backtest_engine.py` | 過濾入場（kwave > 1.5 不入場） |
| `data_pipeline.py` | 出現在 `build_daily_frame` 的 join 結果中 |

在 `feature_engineering.py` 的 `FEATURE_GROUPS` 所有九類中，`kwave_score` 均不存在。在 `config.py` 的 `ALL_FEATURES` 中也不存在。這是整個系統最大的「幽靈特徵」——到處被使用，卻沒有人定義它如何計算。

**建議（明確化的解法）：**
```python
# feature_engineering.py 中明確計算 kwave_score
def add_kwave_score(df: pd.DataFrame) -> pd.DataFrame:
    """
    康波週期評分（簡化代理）：
    使用工業金屬價格/油價比 + 長短端利差 + 勞動生產率的組合
    作為「第幾波、哪個季節」的量化代理
    取值範圍：-2.0（冬季末）~ +2.0（春季初）
    """
    # 目前可用的代理：
    # 正向因子：taiex_rel_strength（市場相對強度）
    # 負向因子：fed_rate_chg_30d（升息代表緊縮）
    # 長短利差：us_yield_spread（倒掛 = 高 kwave 風險）
    if "us_yield_spread" in df.columns and "fed_rate_chg_30d" in df.columns:
        kwave_proxy = (
            df["us_yield_spread"].rolling(252).rank(pct=True) * 2  # 0~2
            - df["fed_rate_chg_30d"].rolling(252).rank(pct=True) * 2  # -2~0
        )
        df["kwave_score"] = kwave_proxy.clip(-2, 2)
    else:
        df["kwave_score"] = 0.0
    return df
```
