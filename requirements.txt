#   sudo apt-get install -y libgomp1 libpq-dev  # OpenMP + PostgreSQL headers

#   xgboost / lightgbm 內含 vcomp140.dll，通常無需額外安裝
#   PostgreSQL：下載官方 installer 或使用 conda 套件
